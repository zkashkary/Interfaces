import 'package:flutter/material.dart'; // Standard Flutter UI package
import 'package:flutter_localizations/flutter_localizations.dart'; // For Arabic RTL support

// 1. START THE APP
void main() {
  runApp(const MyApp()); // Launching the application
}

// 2. MAIN APP CONFIGURATION
class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false, // Hiding the debug tag
      title: 'Fixed Reports',

      // ARABIC RTL SETUP
      // This part handles all the Right-to-Left logic automatically
      localizationsDelegates: const [
        GlobalMaterialLocalizations.delegate,
        GlobalWidgetsLocalizations.delegate,
        GlobalCupertinoLocalizations.delegate,
      ],
      supportedLocales: const [
        Locale('ar', 'SA'), // Supporting Arabic (Saudi Arabia)
      ],
      locale: const Locale('ar', 'SA'), // Forcing the app to start in Arabic

      theme: ThemeData(
        fontFamily: 'Roboto', // Setting the font
      ),
      home: Fixed_Reports(), // Starting the Fixed Reports screen
    );
  }
}

// 3. FIXED REPORTS SCREEN
class Fixed_Reports extends StatelessWidget {
  Fixed_Reports({super.key});

  // THIS IS THE DATA LIST
  // I put the fixed reports here so the counter can read them
  final List<Map<String, dynamic>> _fixedReports = [
    {
      'title': 'المكيف لا يعمل',
      'location': 'مبنى المكتبة - 203',
      'subLocation': 'الطابق الأول، قاعة الدراسة',
      'status': 'تم الإصلاح',
      'category': 'تكييف',
      'date': '٢ فبراير ٢٠٢٦',
      'icon': Icons.ac_unit_outlined, // Snowflake icon
    },
    {
      'title': 'alzahraa kashkary',
      'location': 'مبنى العلوم - 209',
      'subLocation': 'الطابق الثاني، الغرفة 204',
      'status': 'قيد المعالجة',
      'category': 'كهرباء',
      'date': '٣٠ يناير ٢٠٢٦',
      'icon': Icons.bolt_outlined, // Lightning bolt icon
    },

    // Adding more items here will automatically update the "X بلاغ" counter
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: const Color(0xFFE8F0F5), // Light blue background

      // Top Navigation Bar
      appBar: AppBar(
        backgroundColor: Colors.white,
        elevation: 0,
        centerTitle: true,
        leading: IconButton(
          icon: const Icon(Icons.arrow_back, color: Color(0xFF004A8F)),
          onPressed: () => Navigator.of(context).maybePop(), // Back button
        ),
        title: const Text(
          'البلاغات المصلحة', // "Fixed Reports" Title
          style: TextStyle(
            color: Color(0xFF004A8F),
            fontWeight: FontWeight.bold,
            fontSize: 18,
          ),
        ),
      ),

      body: Column(
        crossAxisAlignment: CrossAxisAlignment.start, // Aligning items for RTL
        children: [
          // DYNAMIC COUNTER
          // This uses the length of the list to show the number of reports
          Padding(
            padding: const EdgeInsets.symmetric(horizontal: 20, vertical: 15),
            child: Align(
              alignment: Alignment.centerRight,
              child: Text(
                '${_fixedReports.length} بلاغ',
                style: const TextStyle(
                  color: Colors.grey,
                  fontSize: 14,
                ),
              ),
            ),
          ),

          // LIST OF CARDS
          Expanded(
            child: ListView.builder(
              padding: const EdgeInsets.symmetric(horizontal: 16),
              itemCount: _fixedReports.length,
              itemBuilder: (context, index) {
                final report = _fixedReports[index];
                return _buildFixedReportCard(
                  title: report['title'],
                  location: report['location'],
                  subLocation: report['subLocation'],
                  status: report['status'],
                  category: report['category'],
                  date: report['date'],
                  icon: report['icon'],
                );
              },
            ),
          ),
        ],
      ),
    );
  }

  // 4. HELPER UI FOR THE GREEN CARD
  Widget _buildFixedReportCard({
    required String title,
    required String location,
    required String subLocation,
    required String status,
    required String category,
    required String date,
    required IconData icon,
  }) {
    return Container(
      margin: const EdgeInsets.only(bottom: 16),
      padding: const EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: Colors.white,
        borderRadius: BorderRadius.circular(12),
        boxShadow: [
          BoxShadow(
            color: Colors.black.withOpacity(0.03),
            blurRadius: 10,
            offset: const Offset(0, 4),
          ),
        ],
      ),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          // TOP ROW: Icon, Title, Green Badge
          Row(
            children: [
              Icon(icon, color: const Color(0xFF004A8F), size: 28),
              const SizedBox(width: 12),
              Expanded(
                child: Text(
                  title,
                  style: const TextStyle(
                    fontWeight: FontWeight.bold,
                    fontSize: 16,
                    color: Color(0xFF333333),
                  ),
                ),
              ),
              // THE GREEN BADGE
              Container(
                padding:
                    const EdgeInsets.symmetric(horizontal: 10, vertical: 4),
                decoration: BoxDecoration(
                  color: const Color(0xFF4CAF50), // Green color
                  borderRadius: BorderRadius.circular(6),
                ),
                child: Text(
                  status, // "تم الإصلاح"
                  style: const TextStyle(
                    color: Colors.white,
                    fontSize: 12,
                    fontWeight: FontWeight.bold,
                  ),
                ),
              ),
            ],
          ),

          const SizedBox(height: 12),

          // LOCATION INFO
          Row(
            children: [
              const Icon(Icons.location_on_outlined,
                  size: 16, color: Colors.grey),
              const SizedBox(width: 4),
              Text(location,
                  style: const TextStyle(color: Colors.grey, fontSize: 13)),
            ],
          ),
          const SizedBox(height: 4),
          Row(
            children: [
              const Icon(Icons.apartment_outlined,
                  size: 16, color: Colors.grey),
              const SizedBox(width: 4),
              Text(subLocation,
                  style: const TextStyle(color: Colors.grey, fontSize: 13)),
            ],
          ),

          const Divider(height: 30, thickness: 0.8), // Thin grey line

          // BOTTOM ROW: Category and Date
          Row(
            mainAxisAlignment: MainAxisAlignment.spaceBetween,
            children: [
              Text(
                category,
                style: const TextStyle(
                  color: Color(0xFF007AFF), // Blue for category
                  fontWeight: FontWeight.bold,
                  fontSize: 14,
                ),
              ),
              Text(
                date,
                style: const TextStyle(color: Colors.grey, fontSize: 12),
              ),
            ],
          ),
        ],
      ),
    );
  }
}
