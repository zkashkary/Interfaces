import 'package:flutter/material.dart';
import 'package:flutter_localizations/flutter_localizations.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: 'Report History',
      localizationsDelegates: const [
        GlobalMaterialLocalizations.delegate,
        GlobalWidgetsLocalizations.delegate,
        GlobalCupertinoLocalizations.delegate,
      ],
      supportedLocales: const [
        Locale('ar', 'SA'), // Arabic
      ],
      locale: const Locale('ar', 'SA'),
      theme: ThemeData(
        fontFamily: 'Roboto',
      ),
      home: ReportHistoryScreen(),
    );
  }
}

class ReportHistoryScreen extends StatelessWidget {
  ReportHistoryScreen({super.key});

  // Add or remove items here UI will update the count automatically.
  final List<Map<String, dynamic>> _reports = [
    {
      'title': 'إضاءة معطلة في القاعة',
      'location': 'مبنى العلوم - 209',

        backgroundColor: Colors.white,
        elevation: 0,
        centerTitle: true,
        leading: IconButton(
          icon: const Icon(Icons.arrow_back, color: Color(0xFF004A8F)),

        ],

  }

      child: Column(

        children: [
          Row(

              Icon(icon, color: const Color(0xFF004A8F), size: 28),

                  title,
                  style: const TextStyle(
                    fontWeight: FontWeight.bold,
